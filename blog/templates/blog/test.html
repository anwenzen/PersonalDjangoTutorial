def contact(request):
    if request.method == "POST":
        name = request.POST.get('name')
        message = request.POST.get('message')
        Message(name=name, message=message).save()
        request.method = "GET"
    return render(request, 'blog/contact-us.html', {'message': Message.objects.all()})
